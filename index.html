```html
<header>

<h1>Inschrijven als straatmuzikant</h1>

</header>


<nav>

<a href="index.html">Home</a>

<a href="voorwaarden.html">Voorwaarden</a>

<a href="inschrijving.html">Inschrijven</a>

<a href="deelnemers.html">Deelnemers</a>

</nav>


<main>

<form id="inschrijvingsForm">

<div>

<label for="naam">Naam / Bandnaam *</label>

<input type="text" id="naam" name="naam" required />

</div>

<div>

<label for="email">E-mailadres *</label>

<input type="email" id="email" name="email" required />

</div>

<div>

<label for="genre">Genre / Muziekstijl</label>

<input type="text" id="genre" name="genre" placeholder="bijvoorbeeld folk, jazz, pop" />

</div>

<div>

<label for="omschrijving">Korte omschrijving van jullie act</label>

<textarea id="omschrijving" name="omschrijving" rows="4" placeholder="Vertel iets over je muziek, je instrumenten, etc."></textarea>

</div>

</form>


<div id="melding" aria-live="polite" style="margin-top: 1rem;"></div>

</main>

&copy; 2024 Straatmuzikantenfestival Elst

</footer>


<script>

const form = document.getElementById('inschrijvingsForm');

const melding = document.getElementById('melding');


form.addEventListener('submit', e => {

e.preventDefault();


const naam = form.naam.value.trim();

const email = form.email.value.trim();

const genre = form.genre.value.trim();

const omschrijving = form.omschrijving.value.trim();


if (!naam || !email) {

melding.textContent = 'Vul alstublieft de verplichte velden in.';

melding.style.color = '#b00020';

return;

}


// Valideer eenvoudig het e-mailadres

const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;

if (!emailRegex.test(email)) {

melding.textContent = 'Voer een geldig e-mailadres in.';

melding.style.color = '#b00020';

return;

}


// Haal bestaande deelnemers op uit localStorage

let deelnemers = JSON.parse(localStorage.getItem('deelnemersElst2024')) || [];


// Check of naam of e-mail al bestaat (dubbel inschrijven voorkomen)

const bestaat = deelnemers.some(d => d.email.toLowerCase() === email.toLowerCase() || d.naam.toLowerCase() === naam.toLowerCase());

if (bestaat) {

melding.textContent = 'Deze naam of e-mailadres is al geregistreerd.';

melding.style.color = '#b00020';

return;

}


// Voeg nieuwe deelnemer toe

deelnemers.push({ naam, email, genre, omschrijving });

localStorage.setItem('deelnemersElst2024', JSON.stringify(deelnemers));


melding.textContent = `Bedankt voor je inschrijving, ${naam}! We nemen snel contact met je op.`;

melding.className = 'success';

melding.style.color = '#155724';


form.reset();

});

</script>

</body>

</html>

```

```html
<script>

// Pagina wisselen

function showPage(pageId) {

const pages = ['home', 'voorwaarden', 'inschrijven', 'deelnemers'];

pages.forEach(page => {

document.getElementById(page).style.display = (page === pageId) ? 'block' : 'none';

});

}

showPage('home'); // start home


// Simpele inschrijving-opslag in localStorage (geen backend!)

const formulier = document.getElementById('inschrijfformulier');

const ledenLijst = document.getElementById('deelnemerslijst');

const formBericht = document.getElementById('formbericht');


// Laad deelnemers uit localStorage bij starten

let deelnemers = JSON.parse(localStorage.getItem('straatmuzikantenElst')) || [];


function updateDeelnemerslijst() {

if (deelnemers.length === 0) {

ledenLijst.innerHTML = '<li>Geen inschrijvingen bekend.</li>';

return;

}

ledenLijst.innerHTML = '';

deelnemers.forEach(d => {

const li = document.createElement('li');

li.textContent = `${d.naam} - Genre: ${d.genre} - Instrumenten: ${d.instrumenten}`;

ledenLijst.appendChild(li);

});

}

updateDeelnemerslijst();


formulier.addEventListener('submit', function(e) {

e.preventDefault();


const naam = formulier.naam.value.trim();

const contact = formulier.contact.value.trim();

const genre = formulier.genre.value.trim();

const instrumenten = formulier.instrumenten.value.trim();

const extra = formulier.extra.value.trim();


if (!naam || !contact || !genre || !instrumenten) {

formBericht.style.color = 'red';

formBericht.textContent = 'Vul aub alle verplichte velden in.';

return;

}


const nieuweDeelnemer = {naam, contact, genre, instrumenten, extra};

deelnemers.push(nieuweDeelnemer);

localStorage.setItem('straatmuzikantenElst', JSON.stringify(deelnemers));


formBericht.style.color = 'green';

formBericht.textContent = 'Bedankt voor je inschrijving!';


formulier.reset();

updateDeelnemerslijst();

});

</script>
```
